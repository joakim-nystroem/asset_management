This file exports a single function `computeEditorPosition` that computes the absolute pixel position for FloatingEditor within the translated virtual-chunk:

```typescript
import { createRowController } from '$lib/grid/utils/gridRows.svelte.ts';
import { createColumnController } from '$lib/grid/utils/gridColumns.svelte.ts';
import { createVirtualScrollController } from '$lib/grid/utils/virtualScrollManager.svelte.ts';

// Or whatever the actual export name is — check the file first.

export function computeEditorPosition(
  editRow: number,
  editCol: number,
  editKey: string,
  keys: string[],
  rows: ReturnType<typeof createRowController>,
  columns: ReturnType<typeof createColumnController>,
  virtualScroll: ReturnType<typeof createVirtualScrollController>
): { top: number; left: number; width: number; height: number } {
  // Y: row's absolute position from chunk origin, minus the chunk's translateY shift.
  // rows.getOffsetY(editRow) = cumulative height of rows 0..editRow-1
  // virtualScroll.getOffsetY(rows) = chunk's translateY (rows.getOffsetY(visibleRange.startIndex))
  // Since FloatingEditor is INSIDE the translated chunk, subtract chunk origin to get
  // position relative to the chunk's top edge.
  const chunkOriginY = virtualScroll.getOffsetY(rows);
  const rowAbsoluteY = rows.getOffsetY(editRow);
  const top = rowAbsoluteY - chunkOriginY;

  // X: sum of column widths for columns 0..editCol-1
  let left = 0;
  for (let i = 0; i < editCol; i++) {
    left += columns.getWidth(keys[i]);
  }

  const width = columns.getWidth(editKey);
  const height = rows.getHeight(editRow);

  return { top, left, width, height };
}
```
