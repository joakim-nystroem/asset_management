```svelte
<script lang="ts">
  import type { SafeUser } from '$lib/types';
  import { untrack } from 'svelte';
  import {
    getDataContext,
    getValidationContext,
    getEventQueueContext,
    getChangeControllerContext,
    getHistoryControllerContext,
    getRowGenControllerContext,
    getViewContext,
    getSortContext,
    getEditingContext,
    getChangeContext,
    getColumnContext,
    getSelectionContext,
  } from '$lib/context/gridContext.svelte.ts';
  import { createSelectionController } from '$lib/grid/utils/gridSelection.svelte.ts';
  import { createValidationController } from '$lib/grid/utils/gridValidation.svelte.ts';
  import { createEventHandler } from '$lib/grid/eventQueue/EventHandler.svelte.ts';
  import { createEventQueue } from '$lib/grid/eventQueue/EventQueue.svelte.ts';

  type Props = {
    user: SafeUser | null;
    baseAssets: Record<string, any>[];
    filteredAssets: Record<string, any>[];
    setBaseAssets: (v: Record<string, any>[]) => void;
    setFilteredAssets: (v: Record<string, any>[]) => void;
    locations: any[];
    statuses: any[];
    conditions: any[];
    departments: any[];
  };

  let {
    user, baseAssets, filteredAssets,
    setBaseAssets, setFilteredAssets,
    locations, statuses, conditions, departments,
  }: Props = $props();

  // Read contexts
  const dataCtx = getDataContext();
  const validationCtx = getValidationContext();
  const viewCtx = getViewContext();
  const sortCtx = getSortContext();
  const editingCtx = getEditingContext();
  const changeCtx = getChangeContext();
  const columnCtx = getColumnContext();
  const selectionCtx = getSelectionContext();
  const eventQueueCtx = getEventQueueContext();

  // Controllers (still used until 07-05 eliminates them)
  const changes = getChangeControllerContext();
  const history = getHistoryControllerContext();
  const selection = createSelectionController();
  const rowGen = getRowGenControllerContext();
  const validation = createValidationController();

  // Derived
  const assets = $derived([...filteredAssets, ...rowGen.newRows]);
  const keys = $derived(assets.length > 0 ? Object.keys(assets[0]) : []);

  // Create event system
  const handler = createEventHandler({
    getBaseAssets: () => baseAssets,
    setBaseAssets,
    getFilteredAssets: () => filteredAssets,
    setFilteredAssets,
    dataCtx, viewCtx, sortCtx, editingCtx, changeCtx, columnCtx,
    changes, history, selection, rowGen,
    updateSearchUrl,  // defined below
  });
  const queue = createEventQueue(handler.handle);

  // Publish queue to context so EventListener and action callbacks can access it
  eventQueueCtx.queue = queue;

  // Also publish setters on DataContext for downstream consumers
  dataCtx.setBaseAssets = setBaseAssets;
  dataCtx.setFilteredAssets = setFilteredAssets;

  // --- Synchronous seed ---
  dataCtx.assets = [...filteredAssets, ...rowGen.newRows];
  dataCtx.baseAssets = baseAssets;
  dataCtx.filteredAssetsCount = filteredAssets.length;
  dataCtx.user = user;
  columnCtx.keys = filteredAssets.length > 0 ? Object.keys(filteredAssets[0]) : (baseAssets.length > 0 ? Object.keys(baseAssets[0]) : []);

  // --- Reactive syncs ---
  $effect(() => { dataCtx.assets = assets; });
  $effect(() => { dataCtx.baseAssets = baseAssets; });
  $effect(() => { dataCtx.filteredAssetsCount = filteredAssets.length; });
  $effect(() => { dataCtx.user = user; });
  $effect(() => { columnCtx.keys = keys; });

  // Validation constraints from props
  $effect(() => {
    const locNames = locations.map((l: any) => l.location_name);
    const statNames = statuses.map((s: any) => s.status_name);
    const condNames = conditions.map((c: any) => c.condition_name);
    const deptNames = departments.map((d: any) => d.department_name);
    untrack(() => {
      const constraints = {
        location: locNames, status: statNames,
        condition: condNames, department: deptNames,
      };
      validationCtx.validationConstraints = constraints;
      validation.setConstraints(constraints);
    });
  });

  // Dirty cell tracking
  $effect(() => {
    const allChanges = changes.getAllChanges();
    if (allChanges.length === 0) { selection.clearDirtyCells(); return; }
    const keyMap = new Map(keys.map((key, index) => [key, index]));
    const assetIdMap = new Map(assets.map((asset, index) => [asset.id.toString(), index]));
    const dirtyCells = allChanges
      .map((change) => {
        const row = assetIdMap.get(String(change.id));
        const col = keyMap.get(change.key);
        return { row, col };
      })
      .filter((c) => c.row !== undefined && c.col !== undefined) as { row: number; col: number }[];
    if (dirtyCells.length === 0) selection.clearDirtyCells();
    else selection.setDirtyCells(dirtyCells);
  });

  // --- URL HELPERS (moved from EventListener) ---
  // [Include ALL URL helper code: reactiveUrl, updateSearchUrl, parseUrlFilters, getCurrentUrlState,
  //  uiCtx registration, URL-driven search $effect, handleFilterSelect, handleViewChange,
  //  navigateToError, handleAddNewRow, action callback wiring, logout handler, realtime + cleanup,
  //  filter panel / header menu mutual-close effects, searchManager cleanup]
  // This is a MOVE, not a rewrite — copy every function and $effect from EventListener verbatim,
  // only changing:
  // - `queue.enqueue({ type: 'COMMIT', mode: 'update' })` becomes:
  //   `queue.enqueue({ type: 'COMMIT', mode: 'update', changes: changes.getAllChanges().map(c => ({id: c.id, key: c.key, oldValue: c.oldValue, newValue: c.newValue})), hasInvalidChanges: changes.hasInvalidChanges })`
  // - `queue.enqueue({ type: 'DISCARD' })` becomes:
  //   `queue.enqueue({ type: 'DISCARD', changesToRevert: changes.getAllChanges(true).map(c => ({id: c.id, key: c.key, oldValue: c.oldValue})) })`
  // Everything else copies verbatim.
</script>
```
