  updateSearchUrl: (params: { q?: string; filters?: Filter[]; view?: string }) => void;
};

export function createEventHandler(deps: HandlerDeps): { handle: (event: GridEvent) => Promise<void> } {
  async function handle(event: GridEvent): Promise<void> {
    switch (event.type) {
      case 'COMMIT':
        if (event.mode === 'update') await handleCommitUpdate(event, deps);
        else if (event.mode === 'create') await handleCommitCreate(event, deps);
        break;
      case 'FILTER':
        await handleFilter(event, deps);
        break;
      case 'DISCARD':
        handleDiscard(event, deps);
        break;
      case 'VIEW_CHANGE':
        await handleViewChange(event, deps);
        break;
      case 'WS_DELTA':
        handleWsDelta(event, deps);
        break;
    }
  }
  return { handle };
