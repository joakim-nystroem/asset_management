```typescript
// From CLAUDE.md:
// SHAPE: Array of CellEdit: { row: number | string, col: string, original: string, value: string, valid: boolean }
// row is asset ID (number for existing, "NEW-N" string for new rows). col is column key. [row, col] pair is unique.
// EDIT FLOW: (1) Check edit context — entry exists for [row, col]? (2) If new value === original → remove entry
// (3) Otherwise → validate → upsert entry with new value and valid. This flow runs in FloatingEditor on save.
// CLEARED: On commit (valid entries removed, invalid remain). On discard (all entries removed).
export type CellEdit = {
  row: number | string;  // asset ID for existing, "NEW-N" for new rows
  col: string;           // column key
  original: string;      // original value before any edit
  value: string;         // current dirty value
  valid: boolean;        // whether value passes validation
};

export type EditContext = {
  edits: CellEdit[];     // array of dirty cell edits — entry exists = dirty
};
```

**Create frontend/src/lib/grid/components/new-row/newRow.svelte.ts:**

This is the replacement for rowGeneration.svelte.ts. It exports utility functions that operate on NewRowContext (obtained via `getNewRowContext()`). The logic is extracted directly from `createRowGenerationController()` in rowGeneration.svelte.ts.

```typescript
import { getNewRowContext, getValidationContext } from '$lib/context/gridContext.svelte.ts';

const REQUIRED_FIELDS = new Set([
  'asset_type', 'manufacturer', 'model', 'serial_number', 'wbd_tag',
  'asset_set_type', 'bu_estate', 'department', 'location', 'node', 'status', 'condition',
]);

export function createNewRowUtils() {
  const ctx = getNewRowContext();
  const validCtx = getValidationContext();

  function isValidValue(key: string, value: unknown): boolean {
    const valueStr = String(value ?? '').trim();
    const isEmpty = valueStr === '';
    if (isEmpty && REQUIRED_FIELDS.has(key)) return false;
    const list = validCtx.validationConstraints[key];
    if (!list || list.length === 0) return true;
    if (isEmpty) return true;
    return list.some((v) => v.toLowerCase() === valueStr.toLowerCase());
  }

  function getNewRowId(): string {
    ctx.counter += 1;
    return `NEW-${ctx.counter}`;
  }

  function addNewRows(count: number, template: Record<string, unknown>): Record<string, unknown>[] {
    const created: Record<string, unknown>[] = [];
    for (let i = 0; i < count; i++) {
      const newRow = { ...template, id: getNewRowId() };
      ctx.rows.push(newRow);
      created.push(newRow);
    }
    return created;
  }

  function updateNewRowField(newRowIndex: number, key: string, value: unknown) {
    if (newRowIndex < 0 || newRowIndex >= ctx.rows.length) return;
    if (key === 'id') return;
    ctx.rows[newRowIndex][key] = value;
    const rowErrors = ctx.invalidFields.get(newRowIndex) ?? new Set<string>();
    if (isValidValue(key, value)) {
      rowErrors.delete(key);
    } else {
      rowErrors.add(key);
    }
    if (rowErrors.size === 0) {
      ctx.invalidFields.delete(newRowIndex);
    } else {
      ctx.invalidFields.set(newRowIndex, rowErrors);
    }
    ctx.invalidFields = new Map(ctx.invalidFields); // trigger reactivity
  }

  function deleteNewRow(newRowIndex: number) {
    if (newRowIndex < 0 || newRowIndex >= ctx.rows.length) return;
    ctx.rows.splice(newRowIndex, 1);
    const newInvalidFields = new Map<number, Set<string>>();
    ctx.invalidFields.forEach((fields, idx) => {
      if (idx < newRowIndex) newInvalidFields.set(idx, fields);
      else if (idx > newRowIndex) newInvalidFields.set(idx - 1, fields);
    });
    ctx.invalidFields = newInvalidFields;
  }

  function validateAll(): boolean {
    ctx.invalidFields.clear();
    let allValid = true;
    ctx.rows.forEach((row, rowIndex) => {
      const rowErrors = new Set<string>();
      Object.keys(row).forEach((key) => {
        if (key === 'id') return;
        if (!isValidValue(key, row[key])) {
          rowErrors.add(key);
          allValid = false;
        }
      });
      if (rowErrors.size > 0) ctx.invalidFields.set(rowIndex, rowErrors);
    });
    ctx.invalidFields = new Map(ctx.invalidFields);
    return allValid;
  }

  function isNewRowFieldInvalid(newRowIndex: number, key: string): boolean {
    const fieldErrors = ctx.invalidFields.get(newRowIndex);
    return fieldErrors ? fieldErrors.has(key) : false;
  }

  function clearNewRows() {
    ctx.rows = [];
    ctx.invalidFields = new Map();
    ctx.counter = 0;
  }

  function clearValidation() {
    ctx.invalidFields = new Map();
  }

  return {
    get rows() { return ctx.rows; },
    get hasNewRows() { return ctx.rows.length > 0; },
    get newRowCount() { return ctx.rows.length; },
    get hasInvalidNewRows() { return ctx.invalidFields.size > 0; },
    addNewRows,
    updateNewRowField,
    deleteNewRow,
    validateAll,
    isNewRowFieldInvalid,
    clearNewRows,
    clearValidation,
    isValidValue,
  };
}

export type NewRowUtils = ReturnType<typeof createNewRowUtils>;
```

Note: `createNewRowUtils()` must be called inside a component context (it calls `getNewRowContext()` which requires context). EventHandler receives it via dependency injection.

**`startEdit(row, col, key, currentValue)`** — absorb from gridEdit. Sets editingCtx fields, adjusts column width and row height.

**`save()`** — the unified save flow per CLAUDE.md:
  - Read editingCtx for current edit state (row, col, key, inputValue)
  - Get the asset from dataCtx.assets[editRow]
  - Trim inputValue
  - Reset editingCtx (isEditing=false, etc.), restore column width and row height
  - If newValue === editOriginalValue, return (no change)
  - Optimistic update: `asset[editKey] = newValue`
  - **Unified edit flow for editCtx:**
    - Find existing CellEdit for [asset.id, editKey] in editCtx.edits
    - If newValue === existing.original (or original from asset if no existing entry), REMOVE the entry (reverted to baseline)
    - Otherwise, validate: check `isValidValue(editKey, newValue)` using validationConstraints from validationCtx
    - Upsert: add/update CellEdit with { row: asset.id, col: editKey, original, value: newValue, valid }
  - **History:** Push to historyCtx.undoStack: `[{ id: asset.id, key: editKey, oldValue: editOriginalValue, newValue }]`. Clear redoStack.
  - Return the change info for the caller

**`cancel()`** — absorb from gridEdit. Reset editingCtx, restore column width/row height.

**`undo(assets)`** — absorb from gridHistory:
  - Pop last batch from historyCtx.undoStack
  - For each action in batch (reverse order): find asset by id, set asset[key] = oldValue
  - Push batch to historyCtx.redoStack
  - For each action: run the editCtx upsert flow (the cell now has oldValue — if oldValue === original, remove entry)
  - Return the batch

**`redo(assets)`** — absorb from gridHistory:
  - Pop last batch from historyCtx.redoStack
  - For each action: find asset by id, set asset[key] = newValue
  - Push batch to historyCtx.undoStack
  - For each action: run the editCtx upsert flow
  - Return the batch

**`clearHistory()`** — clears both stacks.

**`clearCommittedHistory(committedEdits)`** — remove committed actions from undoStack.

**`updateRowHeight(textarea)`** — absorb from gridEdit.

**`isValidValue(key, value)`** — absorb from gridValidation/gridChanges. Reads validationConstraints from validationCtx.
