EventQueue.svelte.ts (module-level $state singleton):
```typescript
import { eventHandler } from './EventHandler.svelte';

let queue = $state<any[]>([]);
let isProcessing = $state(false);

export function enqueue(event: any) {
    queue.push(event);
    processNext();
}

async function processNext() {
    if (isProcessing || queue.length === 0) return;
    isProcessing = true;
    const nextEvent = queue.shift();
    try {
        await eventHandler.process(nextEvent);
    } catch (error) {
        console.error("Queue recovered from event failure:", error);
    } finally {
        isProcessing = false;
        processNext();
    }
}
```

Wait — the payload is supposed to be self-contained data, not callbacks. Including `clearChanges()`, `resetSelection()`, `clearNewRows()` in the payload is effectively dependency injection through the event. This violates the spirit of self-contained events.

Better approach: The handler modules import the dataAccessor for assets. For controller operations (changes.clear, selection.reset, rowGen.clearNewRows), the handlers call back to the data accessor pattern OR the EventHandler returns a "result" that EventListener acts on. But per REFERENCE.md, data flows forward only (Listener -> Queue -> Handler -> Target), never backward.

SIMPLEST CORRECT APPROACH: Include callback functions in the event payload. Yes, this looks like DI, but it's not — it's packaging the operation closures into the event at creation time in EventListener. The event is still self-contained because it carries everything needed. This is the pragmatic choice given the temporary controller architecture (07-03/04 eliminate the controllers entirely).

Per the critical directives: "Handler modules receive ALL data they need from the event payload (self-contained events). Handler modules can import singletons (searchManager, realtime, toastState) directly. For things like baseAssets/filteredAssets that the handlers need to read/write: handlers import getter/setter functions from +page.svelte or a thin data accessor module."

So: data read/write via dataAccessor imports, controller operations via event payload closures. This is the correct split.

import { getBaseAssets, setFilteredAssets } from './dataAccessor';
import type { Filter } from '$lib/data/searchManager.svelte';

type FilterPayload = {
  q: string;
  filters: Filter[];
  view: string;
  // Closures packaged by EventListener
  clearChanges: () => void;
  clearHistory: () => void;
  clearNewRows: () => void;
  resetSelection: () => void;

  hasNewRows: boolean;
  changesToRevert: { id: number | string; key: string; oldValue: any }[];

export async function handleFilter(payload: FilterPayload): Promise<void> {
  if (!payload.q && payload.filters.length === 0) {
    setFilteredAssets([...getBaseAssets()]);
    payload.resetSort();
    payload.resetSelection();
    return;
  }

  const params = new URLSearchParams();
  if (payload.q) params.set('q', payload.q);
  payload.filters.forEach(f => params.append('filter', `${f.key}:${f.value}`));
  params.set('view', payload.view);

  try {
    const response = await fetch(`/api/assets?${params.toString()}`);
    if (!response.ok) throw new Error(`API Error: ${response.status}`);
    const result = await response.json();

    if (payload.hasChanges) {
      for (const change of payload.changesToRevert) {
        const item = getBaseAssets().find((a: any) => a.id === change.id);
        if (item) item[change.key] = change.oldValue;
      }
      payload.clearChanges();
      payload.clearHistory();
    }
    if (payload.hasNewRows) payload.clearNewRows();
    setFilteredAssets(result.assets || []);
    payload.resetSelection();
    payload.resetSort();
  } catch (err) {
    console.error('Search failed:', err);
    toastState.addToast("Search failed. Please try again.", "error");
  }
}

// ─── GridEvent type ──────────────────────────────────────────────────────────
// Self-contained event objects. Payload carries ALL data needed by handlers.

export type GridEvent = {
  type: string;
  payload: any;
};

// ─── Module-level singleton ──────────────────────────────────────────────────

let queue = $state<GridEvent[]>([]);
let isProcessing = $state(false);

export function enqueue(event: GridEvent) {
  queue.push(event);
  processNext();
}

async function processNext() {
  if (isProcessing || queue.length === 0) return;

  isProcessing = true;
  const nextEvent = queue.shift()!;

  try {
    await eventHandler.process(nextEvent);
  } catch (error) {
    console.error('[EventQueue] recovered from event failure:', error);
  } finally {
    isProcessing = false;
    processNext();
  }
}
