6. Add sort logic (moved from EventListener):
```typescript
type SortDirection = 'asc' | 'desc';

function sortData<T>(list: T[], key: keyof T, dir: SortDirection): T[] {
  const direction = dir === 'asc' ? 1 : -1;
  return [...list].sort((a, b) => {
    const valA = a[key]; const valB = b[key];
    if (valA == null) return 1;
    if (valB == null) return -1;
    if (typeof valA === 'number' && typeof valB === 'number') return (valA - valB) * direction;
    return String(valA).localeCompare(String(valB)) * direction;
  });
}

async function sortDataAsync(data: any[], key: string, dir: SortDirection): Promise<any[]> {
  return new Promise((resolve) => { setTimeout(() => resolve(sortData(data, key as any, dir)), 0); });
}

async function applySort(key: string, dir: 'asc' | 'desc') {
  // Reset selection (need selectionCtx or a simpler approach)
  // Read baseAssets and filteredAssets from dataCtx
  if (sortCtx.sortKey === key && sortCtx.sortDirection === dir) {
    sortCtx.sortKey = null;
    sortCtx.sortDirection = null;
    dataCtx.setFilteredAssets?.([...dataCtx.baseAssets]);
  } else {
    sortCtx.sortKey = key;
    sortCtx.sortDirection = dir;
    // Sort the current filteredAssets (need to read them)
    // dataCtx.assets includes new rows — sort only the filteredAssets portion
    // We need access to the raw filteredAssets, not the combined assets
    // Use dataCtx.setFilteredAssets with sorted data
    // Get filteredAssets: dataCtx.assets.slice(0, dataCtx.filteredAssetsCount)
    const currentFiltered = dataCtx.assets.slice(0, dataCtx.filteredAssetsCount);
    const sorted = await sortDataAsync(currentFiltered, key, dir);
    dataCtx.setFilteredAssets?.(sorted);
  }
  uiCtx.headerMenu?.close();
}
```
