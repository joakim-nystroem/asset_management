Replaces `clipboardManager`. Clipboard depends on selection state — which is now in gridContext
(from Plan 03). The clipboard controller reads `ctx.selectionStart`, `ctx.selectionEnd`,
`ctx.copyStart`, `ctx.copyEnd`, `ctx.isCopyVisible`, `ctx.isHiddenAfterCopy` from context.

Port the full `clipboardManager` API (copy, paste, clear, etc.) into a factory function that
reads `getGridContext()`. The clipboard buffer (copied cell values) is transient — keep it as
local `$state` inside the controller instance (it's not cross-component state):

```typescript
import { getGridContext } from '$lib/context/gridContext.svelte.ts';

export function createClipboardController() {
  const ctx = getGridContext();

  // Clipboard buffer is local — not needed by other components
  let buffer = $state<string[][]>([]);

  function copy(assets: Record<string, unknown>[], keys: string[]) {
    const minRow = Math.min(ctx.selectionStart.row, ctx.selectionEnd.row);
    const maxRow = Math.max(ctx.selectionStart.row, ctx.selectionEnd.row);
    const minCol = Math.min(ctx.selectionStart.col, ctx.selectionEnd.col);
    const maxCol = Math.max(ctx.selectionStart.col, ctx.selectionEnd.col);

    buffer = [];
    for (let r = minRow; r <= maxRow; r++) {
      const row: string[] = [];
      for (let c = minCol; c <= maxCol; c++) {
        row.push(String(assets[r]?.[keys[c]] ?? ''));
      }
      buffer.push(row);
    }

    // Update context copy overlay state
    ctx.copyStart = { ...ctx.selectionStart };
    ctx.copyEnd = { ...ctx.selectionEnd };
    ctx.isCopyVisible = true;
    ctx.isHiddenAfterCopy = true;
  }

  function paste(assets: Record<string, unknown>[], keys: string[]): Array<{
    assetIndex: number; key: string; oldValue: unknown; newValue: string;
  }> {
    // Port the full paste logic from the original clipboardManager
    // Return array of changes for caller to process/save
    // ...
  }

  function clearCopyOverlay() {
    ctx.isCopyVisible = false;
    ctx.copyStart = { row: -1, col: -1 };
    ctx.copyEnd = { row: -1, col: -1 };
  }

  return { copy, paste, clearCopyOverlay, get bufferSize() { return buffer.length; } };
}

export type ClipboardController = ReturnType<typeof createClipboardController>;
```
